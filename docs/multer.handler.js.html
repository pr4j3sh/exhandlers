 * @example
 * const upload = uploadHandler({ dest: "uploads/" });
 * app.post("/upload", upload.single("file"), (req, res) => {
 *   res.json({ file: req.file });
 * });
 */

  const upload = multer(config);
  return upload;
};

module.exports = { uploadHandler };
